=====<Movelist>=====

-- Button Assign
x - Light Punch
y - Hard Punch
z - Knockdown Attack
a - Light Kick
b - Hard Kick
c - Maximum Mode
start - Taunt


-- Moves
Knockdown Attack - y+b or z
GC Knockdown Attack - y+b or z while guarding
Roll - x+a
GC Roll - x+a while guarding(Requires 1 Power Bar)
Quick Roll - x+a while attack hitting(Requires 1 Power Bar)
Safe Fall Recovery - x+a when nearing ground while falling
Maximum Mode - a+y(Requires 1 Power Bar)
Quick Max - a+y while attack hitting(Requires 2 Power Bars)

-- Throw
Spot Pile - F, y

Knee Strike - F, b


-- Command Moves
One Inch - F, x

Knee Assault - F, a


-- Super Moves
Ein Trigger - D, DF, F, x or y
 ->Second Shoot - F, a
 ->Second Shell - F, b
 ->Black Out    - B, a or b
 ->Second Spike - D, DB, B, a or b

Crow Bites   - F, D, DF, x or y
 ->Follow Up - F, b

Minute Spike    - D, DB, B, a or b(can use in air)
 ->Narrow Spike - D, DB, B, a or b

Black Out - D, DF, F, a or b


-- DM(Normally:Requires 1 Power Bars/MAX Mode:No Cost)
Heat Drive - D, DF, F, D, DF, F, x or y

Chain Drive - D, DF, F, DF, D, DB, B, x or y


-- SDM(Normally:Requires 3 Power Bars/MAX Mode:Requires 1 Power Bars)
Chain Drive - D, DF, F, DF, D, DB, B, x+y


-- MAX2(In low life, Normally:Requires 3 Power Bars/MAX Mode:Requires 1 Power Bars)
Crimson Star Load - D, DB, B, y, x
                  - F, DF, D, DB, B, F, DF, D, DB, B, x+y
